51   /**
52    * simpleScore assumes that we're score an upper case version of the name
53    * Since we're dealing with character values begining with 1, I'm  subtracting
54    * the one less than the int value of 'A' (so I don't have to remember what it is)
55    * so we can get to a value for the character of 'A' of 1
56    *
57    * If some other mapping is desired, this could be modified to accept an arbitrary
58    * mapping like Map[Char,Int]
59    */
60   val simpleScore: (String, Int) => Int = (name: String, sortPosition: Int) => {
61     val chars = name.toList.map(c => (c.toInt - ('A'.toInt - 1)))
62     chars.sum * sortPosition
63   }
